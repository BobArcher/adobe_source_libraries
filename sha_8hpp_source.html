55 template <typename I> // I models InputIterator
56 struct bit_packer
57 {
58  typedef typename std::iterator_traits<I>::value_type value_type;
59 
60  BOOST_STATIC_ASSERT((sizeof(value_type) == 1));
61 
62  bit_packer(I first, I last) :
63  first_m(first), bitsize_m(long_distance(first, last))
64  { }
65 
66  bit_packer(I first, boost::uint64_t bitsize) :
67  first_m(first), bitsize_m(bitsize)
68  { }
69 
70  template <typename T>
71  inline std::size_t operator () (T& result)
72  { return byte_pack(result); }
73 
74 private:
75  template <typename T>
76  std::size_t byte_pack(T& result)
77  {
78  std::size_t to_pack(sizeof(T));
79 
80  result = 0;
81 
82  while (to_pack != 0 && bitsize_m != 0)
83  {
84  std::size_t shift_count(8 * (to_pack - 1));
85  T mask(T(255) << shift_count);
86 
87  result |= (T(*first_m) << shift_count) & mask;
88 
89  if (bitsize_m >= 8)
90  {
91  bitsize_m -= 8;
92  }
93  else
94  {
95  std::size_t remaining(static_cast<std::size_t>(to_pack * 8 - bitsize_m));
96 
97  bitsize_m = 0;
98 
99  return remaining;
100  }
101 
102  ++first_m;
103  --to_pack;
104  }
105 
106  return to_pack * 8;
107  }
108 
109  I first_m;
110  boost::uint64_t bitsize_m;
111 };

197 template <typename HashTraits, typename I>
198 void block_and_digest(typename HashTraits::process_type& digest, I first, boost::uint64_t num_bits)
199 {
200  typedef HashTraits traits_type;
201  typedef typename traits_type::message_block_type message_block_type;
202  typedef typename message_block_type::value_type message_block_value_type;
203 
204  enum
205  {
206  max_message_bitsize_k = traits_type::max_message_bitsize_k,
207  half_max_message_bitsize_k = max_message_bitsize_k / 2,
208  message_blocksize_k = traits_type::message_blocksize_k,
209  use_mb_14 = half_max_message_bitsize_k < bitsizeof<boost::uint64_t>::value
210  };
211 
212  message_block_value_type message_block_value_type_max(std::numeric_limits<message_block_value_type>::max());
213  message_block_type message_block;
214  boost::uint64_t message_size(num_bits + max_message_bitsize_k);
215  boost::uint64_t num_blocks(message_size / message_blocksize_k + 1);
216  bool in_padding(false);
217  bit_packer<I> bits(first, num_bits);
218 
219  while (num_blocks != 0)
220  {
221  for (std::size_t i(0); i < 16; ++i)
222  {
223  if (!in_padding)
224  {
225  std::size_t unset_bits(bits(message_block[i]));
226 
227  if (unset_bits != 0)
228  {
229  message_block[i] |= message_block_value_type(1) << (unset_bits - 1);
230 
231  in_padding = true;
232  }
233  }
234  else
235  {
236  if (num_blocks == 1)
237  {
238  // REVISIT (fbrereto) : According to the SHA standard the message length in the
239  // 1024-block-size case can be up to 2^128 bits long,
240  // but we only support messages up to 2^64 in length. In
241  // all instances when padding in the generic case, block
242  // part 14 would be:
243  // mbp14 = (num_bits >> (half_max_message_bitsize_k)) &
244  // message_block_value_type_max
245  // But in the 1024-block-size case
246  // half_max_message_bitsize_k == num_bits, and we will get
247  // a compiler error basically saying "hey, you're
248  // overshifting this value to 0", which would be fine in
249  // this case because the number should be set to 0, but
250  // the compiler is still (rightfully) noisy about it. This
251  // workaround forces the right thing to do in that it sets
252  // message block part 14 to zero in this special
253  // 1024-block-size case, thus sliencing the compiler.
254 
255  if (i == 14)
256  message_block_part_14_set_t<use_mb_14, traits_type>()(message_block[i], num_bits);
257  else if (i == 15)
258  message_block[i] = static_cast<message_block_value_type>(num_bits & message_block_value_type_max);
259  else
260  message_block[i] = 0;
261  }
262  else
263  message_block[i] = 0;
264  }
265  }
266 
267  traits_type().digest_message_block(digest, message_block);
268 
269  --num_blocks;
270  }
271 
272  // clears potentioally sensitive information
273  std::memset(&message_block, 0, sizeof(message_block));
274 }

278 template <typename HashTraits>
279 void sha_2_digest_message_block(typename HashTraits::process_type& digest,
280  const typename HashTraits::message_block_type& message_block)
281 {
282  // The "sha_2" in the name of this function is in
283  // reference to the second generation of SHA algorithms
284  // (224, 256, 384, and 512), all of which have the same
285  // message block process implementation.
286 
287  typedef HashTraits traits_type;
288  typedef typename traits_type::message_block_type message_block_type;
289  typedef typename traits_type::schedule_type schedule_type;
290  typedef typename traits_type::process_type::value_type digest_value_type;
291 
292  schedule_type schedule;
293 
294  adobe::copy(message_block, &schedule[0]);
295 
296  for (std::size_t t(message_block_type::static_size); t < schedule_type::static_size; ++t)
297  schedule[t] = traits_type::small_sigma_1(schedule[t - 2]) + schedule[t - 7] +
298  traits_type::small_sigma_0(schedule[t - 15]) + schedule[t - 16];
299 
300  digest_value_type a(digest[0]);
301  digest_value_type b(digest[1]);
302  digest_value_type c(digest[2]);
303  digest_value_type d(digest[3]);
304  digest_value_type e(digest[4]);
305  digest_value_type f(digest[5]);
306  digest_value_type g(digest[6]);
307  digest_value_type h(digest[7]);
308 
309  for (std::size_t t(0); t < schedule.size(); ++t)
310  {
311  digest_value_type T1 = h +
312  traits_type::big_sigma_1(e) +
313  implementation::ch(e, f, g) +
314  traits_type::k(t) +
315  schedule[t];
316  digest_value_type T2 = traits_type::big_sigma_0(a) +
317  implementation::maj(a, b, c);
318  h = g;
319  g = f;
320  f = e;
321  e = d + T1;
322  d = c;
323  c = b;
324  b = a;
325  a = T1 + T2;
326  }
327 
328  digest[0] += a;
329  digest[1] += b;
330  digest[2] += c;
331  digest[3] += d;
332  digest[4] += e;
333  digest[5] += f;
334  digest[6] += g;
335  digest[7] += h;
336 
337  // clears potentioally sensitive information
338  std::memset(&schedule, 0, sizeof(schedule));
339 }

373  schedule[t - 14] ^ schedule[t - 16]);

376  boost::uint32_t b(digest[1]);
377  boost::uint32_t c(digest[2]);

380

382  {
383  boost::uint32_t T = implementation::rotl<5>(a) +
384  f(t, b, c, d) +
385  e +
386  k(t) +
387  schedule[t];

389  d = c;
390  c = implementation::rotl<30>(b);

392  a = T;
393  }
394

396  digest[1] += b;
397  digest[2] += c;
398  digest[3] += d;
399  digest[4] += e;
400 
401  // clears potentioally sensitive information
402  std::memset(&schedule, 0, sizeof(schedule));
403  }
